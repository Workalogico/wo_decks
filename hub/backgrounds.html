CSS - Estilos requeridos 
#tsparticles {
  position: fixed;
  top: 0; left: 0;
  width: 100%; height: 100%;
  z-index: 0;
  pointer-events: none;
  opacity: 0;
  transition: opacity 0.8s ease;
}

#tsparticles.visible { opacity: 1; }

.reveal { position: relative; z-index: 1; }

/* Atenuar grid cuando hay partículas activas */
.particles-active .grid-background { opacity: 0.3 !important; }
JavaScript - Configuración tsParticles 
const particlesConfig = {
  particles: {
    color: { value: ["#5968EA", "#5968EA", "#5968EA", "#FFCB00"] }, // 75% azul, 25% amarillo
    links: {
      enable: true,
      color: "#5968EA",
      distance: 140,
      opacity: 0.12,
      width: 1
    },
    move: {
      enable: true,
      speed: 0.6,
      direction: "none",
      outModes: { default: "out" }
    },
    number: { value: 50 },
    opacity: { value: { min: 0.3, max: 0.7 } },
    size: { value: { min: 2, max: 4 } }
  },
  interactivity: {
    events: {
      onHover: { enable: true, mode: "grab" }
    },
    modes: {
      grab: {
        distance: 180,
        links: { color: "#FFCB00", opacity: 0.5 } // Amarillo al hover
      }
    }
  }
};

// Inicializar
tsParticles.load("tsparticles", particlesConfig);
HTML - Activar en slide 
<!-- Agregar contenedor antes de .reveal -->
<div id="tsparticles"></div>

<!-- Activar partículas con data-attribute -->
<section data-background-gradient="radial-gradient(...)" 
         data-particles="true">
  <div class="wo-cover">...</div>
</section>

<!-- Control de visibilidad en Reveal.js -->
<script>
Reveal.on('slidechanged', event => {
  const particles = document.getElementById('tsparticles');
  const hasParticles = event.currentSlide.dataset.particles === 'true';
  particles.classList.toggle('visible', hasParticles);
  document.body.classList.toggle('particles-active', hasParticles);
});
</script>

Fondo interactivo 3D con aviones de papel animados usando Three.js y el algoritmo de flocking (boids). Los aviones siguen el cursor y crean un efecto dinámico.

THREE.JS + WEBGL
Flocking Simulation
Mueve el cursor para interactuar con los aviones de papel
