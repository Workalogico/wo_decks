function startGeoDemo() {
  const queryText = document.getElementById('geo-query-text');
  const responseEl = document.getElementById('geo-response');
  const responseText = document.getElementById('geo-response-text');
  const aiIcons = document.querySelectorAll('.geo-ai-icon');
  const cursor = document.querySelector('.geo-cursor');
  
  const query = "¬øMejores desarrollos inmobiliarios en la costa?";
  const response = "Miramar Hills destaca como opci√≥n premium: ubicaci√≥n privilegiada, arquitectura contempor√°nea y amenidades de primer nivel...";
  
  // Fase 1: Typing de la pregunta
  let charIndex = 0;
  const typingInterval = setInterval(() => {
    if (charIndex < query.length) {
      queryText.textContent = query.substring(0, charIndex + 1);
      charIndex++;
    } else {
      clearInterval(typingInterval);
      cursor.style.display = 'none';
      
      // Fase 2: Activar iconos de IA
      setTimeout(() => {
        aiIcons.forEach((icon, index) => {
          setTimeout(() => icon.classList.add('active'), index * 200);
        });
        
        // Fase 3: Mostrar respuesta
        setTimeout(() => {
          responseEl.classList.add('visible');
          
          // Typing de la respuesta
          let respCharIndex = 0;
          const respInterval = setInterval(() => {
            if (respCharIndex < response.length) {
              responseText.textContent = response.substring(0, respCharIndex + 1);
              respCharIndex++;
            } else {
              clearInterval(respInterval);
            }
          }, 25);
        }, 800);
      }, 400);
    }
  }, 50);
}

JavaScript 
function startSemDemo() {
  const metrics = document.querySelectorAll('.sem-metric');
  const progressBar = document.getElementById('sem-progress');
  const clicksEl = document.querySelector('#sem-metric-clicks .sem-metric-value');
  const ctrEl = document.querySelector('#sem-metric-ctr .sem-metric-value');
  const convEl = document.querySelector('#sem-metric-conv .sem-metric-value');
  
  // Mostrar m√©tricas con stagger
  metrics.forEach((metric, index) => {
    setTimeout(() => metric.classList.add('visible'), index * 200);
  });
  
  // Animar contadores
  setTimeout(() => {
    animateCounter(clicksEl, 0, 2847, 1500, '');
    animateCounter(ctrEl, 0, 4.2, 1500, '%');
    animateCounter(convEl, 0, 156, 1500, '');
    
    if (progressBar) {
      progressBar.style.width = '87%';
    }
  }, 600);
}

function animateCounter(element, start, end, duration, suffix) {
  const startTime = performance.now();
  const isDecimal = !Number.isInteger(end);
  
  function update(currentTime) {
    const elapsed = currentTime - startTime;
    const progress = Math.min(elapsed / duration, 1);
    const easeProgress = 1 - Math.pow(1 - progress, 3); // ease-out cubic
    
    const current = start + (end - start) * easeProgress;
    element.textContent = (isDecimal ? current.toFixed(1) : Math.floor(current)) + suffix;
    
    if (progress < 1) requestAnimationFrame(update);
  }
  
  requestAnimationFrame(update);
}
