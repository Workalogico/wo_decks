Instalación

El componente de pricing es modular. Importa CSS y JS según necesites.

HTML 
<!-- CSS (requerido) -->
<link rel="stylesheet" href="../wo-components/wo-pricing.css">

<!-- JS (opcional, para animaciones) -->
<script src="../wo-components/wo-pricing.js"></script>

Pricing Grid

Contenedor con align-items: stretch para altura uniforme.

CSS 
.wo-grid--pricing {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: 1.8em;
  max-width: 1000px;
  margin: 0 auto;
  align-items: stretch; /* Altura uniforme */
}

Pricing Card - Estructura

Estructura interna con 3 secciones: pricing-header, pricing-value, pricing-footer.

HTML 
<div class="wo-card wo-card--pricing" data-animate>
  <div class="pricing-header">
    <h3>Iguala Mensual</h3>
    <p class="card-tier">Pago recurrente</p>
  </div>
  <div class="pricing-value">
    <p class="card-price" data-price="19500">$19,500</p>
    <p class="card-time">+IVA / mes</p>
  </div>
  <div class="pricing-footer pricing-footer--empty"></div>
</div>

Secciones Internas

Clase Propósito
.pricing-header Contiene título (h3) y subtítulo (.card-tier)
.pricing-value Precio base, precio actual, IVA. Usa flex: 1 para ocupar espacio.
.pricing-footer Descuento y ahorro. Altura mínima 50px para uniformidad.
.pricing-footer--empty Placeholder para cards sin descuento (mismo espacio).

JavaScript de Animación

Función para iniciar las animaciones cuando el slide se hace visible.

JavaScript 
function startPricingAnimation() {
  const pricingCards = document.querySelectorAll('.wo-card--pricing');
  const discounts = document.querySelectorAll('.wo-pricing-discount');
  const savings = document.querySelectorAll('.wo-pricing-savings');
  const basePrices = document.querySelectorAll('.wo-pricing-base');
  
  // Fase 1: Entrada escalonada de cards (stagger 150ms)
  pricingCards.forEach((card, index) => {
    setTimeout(() => {
      card.classList.add('animate-in');
      
      // Animar contador de precio
      const priceEl = card.querySelector('.card-price');
      if (priceEl) {
        setTimeout(() => animatePriceCounter(priceEl), 300);
      }
    }, index * 150);
  });
  
  // Fase 2: Mostrar precios base (tachados)
  setTimeout(() => {
    basePrices.forEach(base => base.classList.add('visible'));
  }, 700);
  
  // Fase 3: Revelar descuentos
  setTimeout(() => {
    discounts.forEach((discount, index) => {
      setTimeout(() => discount.classList.add('visible'), index * 200);
    });
  }, 1000);
  
  // Fase 4: Mostrar ahorros
  setTimeout(() => {
    savings.forEach((saving, index) => {
      setTimeout(() => saving.classList.add('visible'), index * 150);
    });
  }, 1400);
}

function animatePriceCounter(element) {
  const targetPrice = parseInt(element.dataset.price) || 0;
  const duration = 1200;
  const startTime = performance.now();
  
  function formatPrice(num) {
    return '$' + Math.floor(num).toLocaleString('en-US');
  }
  
  function update(currentTime) {
    const elapsed = currentTime - startTime;
    const progress = Math.min(elapsed / duration, 1);
    const easeProgress = 1 - Math.pow(1 - progress, 4); // ease-out expo
    
    const currentValue = targetPrice * easeProgress;
    element.textContent = formatPrice(currentValue);
    
    if (progress < 1) {
      requestAnimationFrame(update);
    } else {
      element.textContent = formatPrice(targetPrice);
    }
  }
  
  element.textContent = '$0';
  requestAnimationFrame(update);
}

<script>
// Activar animación al llegar al slide
Reveal.on('slidechanged', (event) => {
  if (event.currentSlide.dataset.state === 'pricing-slide') {
    setTimeout(() => startPricingAnimation(), 300);
  }
});
</script>
