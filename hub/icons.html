Código de Íconos Animados

Código HTML y CSS para integrar los íconos animados en presentaciones.

HTML - Estructura 
<!-- Card con ícono animado -->
<div class="wo-card wo-card--icon">
  <div class="wo-card__icon" data-icon="restaurant">
    <svg viewBox="0 0 48 48">
      <path class="icon-path" d="M12 4v16c0 2.2..."/>
    </svg>
  </div>
  <h3>Título</h3>
  <p>Descripción</p>
</div>

<!-- Ícono standalone -->
<div class="wo-icon wo-icon--lg wo-icon--yellow" data-animate="draw">
  <svg viewBox="0 0 48 48">
    <path class="icon-path" d="..."/>
  </svg>
</div>
JavaScript - Animación con Anime.js 
// Animar íconos con path drawing
const icons = document.querySelectorAll('.wo-card__icon');

icons.forEach((icon, index) => {
  icon.classList.add('animated');
  const paths = icon.querySelectorAll('.icon-path');
  const dots = icon.querySelectorAll('.icon-dot');
  
  anime({
    targets: paths,
    strokeDashoffset: [200, 0],
    opacity: [0, 1],
    easing: 'easeOutQuad',
    duration: 800,
    delay: 300 + (index * 100)
  });
  
  if (dots.length > 0) {
    anime({
      targets: dots,
      scale: [0, 1],
      opacity: [0, 1],
      easing: 'easeOutBack',
      duration: 400,
      delay: 800 + (index * 100)
    });
  }
});
